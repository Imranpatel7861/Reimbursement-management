CREATE TABLE companies (
    id SERIAL PRIMARY KEY,
    name VARCHAR(150) NOT NULL,
    currency VARCHAR(10) NOT NULL,
    country VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    password TEXT NOT NULL,

    role VARCHAR(20) NOT NULL CHECK (
        role IN ('ADMIN','EMPLOYEE','MANAGER','FINANCE','DIRECTOR')
    ),

    company_id INTEGER NOT NULL,
    manager_id INTEGER,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_company
        FOREIGN KEY (company_id)
        REFERENCES companies(id)
        ON DELETE CASCADE,

    CONSTRAINT fk_manager
        FOREIGN KEY (manager_id)
        REFERENCES users(id)
        ON DELETE SET NULL
);


CREATE TABLE expenses (
    id SERIAL PRIMARY KEY,

    employee_id INTEGER NOT NULL,
    company_id INTEGER NOT NULL,

    amount NUMERIC(10,2) NOT NULL,
    currency VARCHAR(10) NOT NULL,

    category VARCHAR(100),
    description TEXT,

    expense_date DATE,

    status VARCHAR(20) DEFAULT 'PENDING' CHECK (
        status IN ('PENDING','APPROVED','REJECTED')
    ),

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_employee
        FOREIGN KEY (employee_id)
        REFERENCES users(id)
        ON DELETE CASCADE,

    CONSTRAINT fk_company_exp
        FOREIGN KEY (company_id)
        REFERENCES companies(id)
        ON DELETE CASCADE
);

CREATE TABLE approval_workflow (
    id SERIAL PRIMARY KEY,

    expense_id INTEGER NOT NULL,
    approver_id INTEGER NOT NULL,

    role VARCHAR(20),

    sequence INTEGER NOT NULL,

    status VARCHAR(20) DEFAULT 'PENDING' CHECK (
        status IN ('PENDING','APPROVED','REJECTED')
    ),

    comment TEXT,

    action_date TIMESTAMP,

    CONSTRAINT fk_expense
        FOREIGN KEY (expense_id)
        REFERENCES expenses(id)
        ON DELETE CASCADE,

    CONSTRAINT fk_approver
        FOREIGN KEY (approver_id)
        REFERENCES users(id)
        ON DELETE CASCADE
);

CREATE TABLE approval_rules (
    id SERIAL PRIMARY KEY,

    company_id INTEGER NOT NULL,

    rule_type VARCHAR(50),

    percentage_required INTEGER,

    specific_approver_role VARCHAR(20),

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_company_rule
        FOREIGN KEY (company_id)
        REFERENCES companies(id)
        ON DELETE CASCADE
);
